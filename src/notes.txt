Kodeeksempler: Interface.
i java: 
interface Animal {
    void makeSound();
}
class Dog implements Animal {
    public void makeSound() {
        System.out.println("Woof!");
    }
}

i c++:
#include <iostream>
using namespace std;

class Animal {
public:
    virtual void makeSound() = 0; // Ren virtuell funksjon
    virtual ~Animal() {} // Viktig: Virtuell destrukt√∏r for korrekt nedarving
};

class Dog : public Animal {
public:
    void makeSound() override {
        cout << "Woof!" << endl;
    }
};

int main() {
    Animal* myDog = new Dog();
    myDog->makeSound();  // Output: Woof!
    delete myDog;
    return 0;
}
